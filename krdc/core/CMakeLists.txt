
add_definitions(-DKDE_DEFAULT_DEBUG_AREA=5010)

set(krdccore_SRCS
    remoteviewfactory.cpp
    remoteview.cpp
    hostpreferences.cpp
)

install(TARGETS krdccore ${INSTALL_TARGETS_DEFAULT_ARGS})

# Install headers
set(krdccore_HDRS
    remoteviewfactory.h
    remoteview.h
    hostpreferences.h
)

install(FILES ${krdccore_HDRS} DESTINATION ${INCLUDE_INSTALL_DIR}/krdc COMPONENT Devel)
